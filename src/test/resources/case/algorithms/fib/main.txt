class Fib {
    fn new() super()

    fn fib(n: u32): u32 {
        if n < 2 {
            n
        } else {
            this.fib(n-1) + this.fib(n-2)
        }
    }
}

Test.assertEqualsU(1134903170, new Fib().fib(45))
